void setup() {
    int x = 300;
    int y = 300;
    int vx = -1;
    int vy = 1;
}
void draw() {
    rect(x, y, 300, 200);
    x = x+vx;
    y = y+vy;
    if ( x < 0 ) {
        vx = -1*vx;
    }
    if ( y > 650 ) {
        vy = -1*vy;
    }
    if ( x > 600 ) {
        vx = -1*vx;
    }
    if ( y < 0 ) {
        vy = -1*vy;
    }
}
